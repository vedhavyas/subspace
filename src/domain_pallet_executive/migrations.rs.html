1//! Runtime storage migrations for pallet-executive and system pallets.
2
3use event_segments_migration::EventSegments;
4use frame_support::traits::{Get, OnRuntimeUpgrade};
5use frame_support::weights::Weight;
6use frame_system::pallet::Config;
7use frame_system::{EventRecord, Pallet as System};
8
9pub struct StorageCheckedMigrateToEventSegments<T>(sp_std::marker::PhantomData<T>);
10
11impl<T: Config> OnRuntimeUpgrade for StorageCheckedMigrateToEventSegments<T> {
12    fn on_runtime_upgrade() -> Weight {
13        // The system pallet doesn't have storage versions, so we need to check if we were previously using Events storage instead.
14        // The migrations run after the upgraded System::reset_events() runs on the old storage format, which leads to an inconsistency:
15        // - the storage claims there are uncleared events, because it was set to the old event count, but
16        // - the event segments storage is empty.
17        //
18        // This migration is safe to run on any upgrade, because:
19        // - if event segments are disabled (or there are no uncleared events for any other reason), it will do nothing;
20        // - if event segments are enabled and in use, it will also do nothing.
21        if System::<T>::uncleared_event_count() > 0 && EventSegments::<T>::get(0).is_none() {
22            let weights = frame_system::migrations::migrate_from_events_to_event_segments::<T>();
23
24            weights + T::DbWeight::get().reads(2)
25        } else {
26            // Checking if we need to migrate costs up to 2 reads
27            T::DbWeight::get().reads(2)
28        }
29    }
30}

32pub(super) mod event_segments_migration {
33    use super::{Config, EventRecord, System};
34    use frame_support::pallet_prelude::{OptionQuery, ValueQuery};
35    use frame_support::{storage_alias, Identity};
36    #[cfg(not(feature = "std"))]
37    use sp_std::boxed::Box;
38    #[cfg(not(feature = "std"))]
39    use sp_std::vec::Vec;
40    // frame_system::EventIndex is a private type
41    use core::primitive::u32 as EventIndex;
42
43    #[storage_alias]
44    pub(super) type Events<T: Config> = StorageValue<
45        System<T>,
46        Vec<
47            Box<
48                EventRecord<
49                    <T as frame_system::Config>::RuntimeEvent,
50                    <T as frame_system::Config>::Hash,
51                >,
52            >,
53        >,
54        ValueQuery,
55    >;
56
57    #[storage_alias]
58    pub(super) type EventSegments<T: Config> = StorageMap<
59        System<T>,
60        Identity,
61        EventIndex,
62        Vec<
63            Box<
64                EventRecord<
65                    <T as frame_system::Config>::RuntimeEvent,
66                    <T as frame_system::Config>::Hash,
67                >,
68            >,
69        >,
70        OptionQuery,
71    >;
72
73    #[storage_alias]
74    pub(super) type UnclearedEventCount<T: Config> =
75        StorageValue<System<T>, EventIndex, ValueQuery>;
76}

78#[cfg(test)]
79mod tests {
80    use super::event_segments_migration::{Events, UnclearedEventCount};
81    use super::*;
82    use crate::mock::{new_test_ext, MockRuntime, RuntimeEvent};
83    use frame_support::weights::RuntimeDbWeight;
84
85    #[test]
86    fn test_system_event_segments_storage_migration() {
87        let event = Box::new(EventRecord {
88            phase: Default::default(),
89            event: RuntimeEvent::System(frame_system::Event::CodeUpdated),
90            topics: Default::default(),
91        });
92
93        // Migration needed: "uncleared" events, but no event segments
94        let mut ext = new_test_ext();
95        ext.execute_with(|| {
96            UnclearedEventCount::<MockRuntime>::put(1);
97        });
98        ext.commit_all().unwrap();
99        ext.execute_with(|| {
100            let weights = StorageCheckedMigrateToEventSegments::<MockRuntime>::on_runtime_upgrade();
101
102            // 2 reads and 2 writes: check then clear Events and UnclearedEventCount
103            let db_weights: RuntimeDbWeight =
104                <MockRuntime as frame_system::Config>::DbWeight::get();
105            assert_eq!(weights, db_weights.reads_writes(2, 2));
106
107            assert_eq!(Events::<MockRuntime>::get(), Vec::new());
108            assert_eq!(EventSegments::<MockRuntime>::get(0), None);
109            assert_eq!(UnclearedEventCount::<MockRuntime>::get(), 0);
110        });
111
112        // Migration not needed: no "uncleared" events (and no event segments)
113        let mut ext = new_test_ext();
114        ext.execute_with(|| {
115            let weights = StorageCheckedMigrateToEventSegments::<MockRuntime>::on_runtime_upgrade();
116
117            // 2 reads: check Events and UnclearedEventCount
118            let db_weights: RuntimeDbWeight =
119                <MockRuntime as frame_system::Config>::DbWeight::get();
120            assert_eq!(weights, db_weights.reads_writes(2, 0));
121
122            assert_eq!(Events::<MockRuntime>::get(), Vec::new());
123            assert_eq!(EventSegments::<MockRuntime>::get(0), None);
124            assert_eq!(UnclearedEventCount::<MockRuntime>::get(), 0);
125        });
126
127        // Migration not needed: event segments already in use (and uncleared events)
128        let mut ext = new_test_ext();
129        ext.execute_with(|| {
130            EventSegments::<MockRuntime>::set(0, Some(vec![event.clone()]));
131            UnclearedEventCount::<MockRuntime>::put(1);
132        });
133        ext.commit_all().unwrap();
134        ext.execute_with(|| {
135            let weights = StorageCheckedMigrateToEventSegments::<MockRuntime>::on_runtime_upgrade();
136
137            // 2 reads: check Events and UnclearedEventCount
138            let db_weights: RuntimeDbWeight =
139                <MockRuntime as frame_system::Config>::DbWeight::get();
140            assert_eq!(weights, db_weights.reads_writes(2, 0));
141
142            assert_eq!(Events::<MockRuntime>::get(), Vec::new());
143            assert_eq!(
144                EventSegments::<MockRuntime>::get(0),
145                Some(vec![event.clone()])
146            );
147            assert_eq!(UnclearedEventCount::<MockRuntime>::get(), 1);
148        });
149
150        // Migration not needed: event segments already in use (and no uncleared events)
151        let mut ext = new_test_ext();
152        ext.execute_with(|| {
153            EventSegments::<MockRuntime>::set(0, Some(vec![event.clone()]));
154        });
155        ext.commit_all().unwrap();
156        ext.execute_with(|| {
157            let weights = StorageCheckedMigrateToEventSegments::<MockRuntime>::on_runtime_upgrade();
158
159            // 2 reads: check Events and UnclearedEventCount
160            let db_weights: RuntimeDbWeight =
161                <MockRuntime as frame_system::Config>::DbWeight::get();
162            assert_eq!(weights, db_weights.reads_writes(2, 0));
163
164            assert_eq!(Events::<MockRuntime>::get(), Vec::new());
165            assert_eq!(EventSegments::<MockRuntime>::get(0), Some(vec![event]));
166            assert_eq!(UnclearedEventCount::<MockRuntime>::get(), 0);
167        });
168    }
