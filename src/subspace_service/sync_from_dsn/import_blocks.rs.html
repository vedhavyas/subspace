22/// How many blocks to queue before pausing and waiting for blocks to be imported, this is
23/// essentially used to ensure we use a bounded amount of RAM during sync process.
24const QUEUED_BLOCKS_LIMIT: BlockNumber = 500;
25/// Time to wait for blocks to import if import is too slow
26const WAIT_FOR_BLOCKS_TO_IMPORT: Duration = Duration::from_secs(1);
27
28/// Starts the process of importing blocks.
29///
30/// Returns number of downloaded blocks.
31#[allow(clippy::too_many_arguments)]
32pub(super) async fn import_blocks_from_dsn<Block, AS, Client, PG, IQS>(
33    segment_headers_store: &SegmentHeadersStore<AS>,
34    segment_header_downloader: &SegmentHeaderDownloader<'_>,
35    client: &Client,
36    piece_getter: &PG,
37    import_queue_service: &mut IQS,
38    last_processed_segment_index: &mut SegmentIndex,
39    last_processed_block_number: &mut <Block::Header as Header>::Number,
40    erasure_coding: &ErasureCoding,
41) -> Result<u64, Error>
42where
43    Block: BlockT,
44    AS: AuxStore + Send + Sync + 'static,
45    Client: HeaderBackend<Block> + BlockBackend<Block> + Send + Sync + 'static,
46    PG: PieceGetter,
47    IQS: ImportQueueService<Block> + ?Sized,
48{
49    {
50        let last_segment_header = segment_headers_store.last_segment_header().ok_or_else(|| {
51            Error::Other(
52                "Archiver needs to be initialized before syncing from DSN to populate the very \
53                first segment"
54                    .to_string(),
55            )
56        })?;
57
58        let new_segment_headers = segment_header_downloader
59            .get_segment_headers(&last_segment_header)
60            .await
61            .map_err(|error| error.to_string())?;
62
63        debug!("Found {} new segment headers", new_segment_headers.len());
64
65        if !new_segment_headers.is_empty() {
66            segment_headers_store.add_segment_headers(&new_segment_headers)?;
67        }
68    }
69
70    let mut imported_blocks = 0;
71    let mut reconstructor = Arc::new(Mutex::new(Reconstructor::new(erasure_coding.clone())));
72    // Start from the first unprocessed segment and process all segments known so far
73    let segment_indices_iter = (*last_processed_segment_index + SegmentIndex::ONE)
74        ..=segment_headers_store
75            .max_segment_index()
76            .expect("Exists, we have inserted segment headers above; qed");
77    let mut segment_indices_iter = segment_indices_iter.peekable();
78
79    while let Some(segment_index) = segment_indices_iter.next() {
80        debug!(%segment_index, "Processing segment");
81
82        let segment_header = segment_headers_store
83            .get_segment_header(segment_index)
84            .expect("Statically guaranteed to exist, see checks above; qed");
85
86        let last_archived_block_number = segment_header.last_archived_block().number;
87        let last_archived_block_partial = segment_header
88            .last_archived_block()
89            .archived_progress
90            .partial()
91            .is_some();
92
93        trace!(
94            %segment_index,
95            last_archived_block_number,
96            last_archived_block_partial,
97            "Checking segment header"
98        );
99
100        let last_archived_block_number = NumberFor::<Block>::from(last_archived_block_number);
101
102        let info = client.info();
103        // We have already processed this block, it can't change
104        if last_archived_block_number <= *last_processed_block_number {
105            *last_processed_segment_index = segment_index;
106            // Reset reconstructor instance
107            reconstructor = Arc::new(Mutex::new(Reconstructor::new(erasure_coding.clone())));
108            continue;
109        }
110        // Just one partial unprocessed block and this was the last segment available, so nothing to
111        // import
112        if last_archived_block_number == *last_processed_block_number + One::one()
113            && last_archived_block_partial
114            && segment_indices_iter.peek().is_none()
115        {
116            // Reset reconstructor instance
117            reconstructor = Arc::new(Mutex::new(Reconstructor::new(erasure_coding.clone())));
118            continue;
119        }
120
121        let segment_pieces = download_segment_pieces(segment_index, piece_getter)
122            .await
123            .map_err(|error| format!("Failed to download segment pieces: {error}"))?;
124        // CPU-intensive piece and segment reconstruction code can block the async executor.
125        let segment_contents_fut = task::spawn_blocking({
126            let reconstructor = reconstructor.clone();
127
128            move || {
129                reconstructor
130                    .lock()
131                    .expect("Panic if previous thread panicked when holding the mutex")
132                    .add_segment(segment_pieces.as_ref())
133            }
134        });
135        let blocks = segment_contents_fut
136            .await
137            .expect("Panic if blocking task panicked")
138            .map_err(|error| error.to_string())?
139            .blocks;
140        trace!(%segment_index, "Segment reconstructed successfully");
141
142        let mut blocks_to_import = Vec::with_capacity(QUEUED_BLOCKS_LIMIT as usize);
143
144        let mut best_block_number = info.best_number;
145        for (block_number, block_bytes) in blocks {
146            let block_number = block_number.into();
147            if block_number == 0u32.into() {
148                let signed_block = client
149                    .block(
150                        client
151                            .hash(block_number)?
152                            .expect("Block before best block number must always be found; qed"),
153                    )?
154                    .expect("Block before best block number must always be found; qed");
155
156                if encode_block(signed_block) != block_bytes {
157                    return Err(Error::Other(
158                        "Wrong genesis block, block import failed".to_string(),
159                    ));
160                }
161            }
162
163            // Limit number of queued blocks for import
164            // NOTE: Since best block number might be non-canonical, we might actually have more
165            // than `QUEUED_BLOCKS_LIMIT` elements in the queue, but it should be rare and
166            // insignificant. Feel free to address this in case you have a good strategy, but it
167            // seems like complexity is not worth it.
168            while block_number.saturating_sub(best_block_number) >= QUEUED_BLOCKS_LIMIT.into() {
169                if !blocks_to_import.is_empty() {
170                    // Import queue handles verification and importing it into the client
171                    import_queue_service
172                        .import_blocks(BlockOrigin::NetworkInitialSync, blocks_to_import.clone());
173                    blocks_to_import.clear();
174                }
175                trace!(
176                    %block_number,
177                    %best_block_number,
178                    "Number of importing blocks reached queue limit, waiting before retrying"
179                );
180                tokio::time::sleep(WAIT_FOR_BLOCKS_TO_IMPORT).await;
181                best_block_number = client.info().best_number;
182            }
183
184            let signed_block =
185                decode_block::<Block>(&block_bytes).map_err(|error| error.to_string())?;
186
187            *last_processed_block_number = last_archived_block_number;
188
189            // No need to import blocks that are already present, if block is not present it might
190            // correspond to a short fork, so we need to import it even if we already have another
191            // block at this height
192            if client.expect_header(signed_block.block.hash()).is_ok() {
193                continue;
194            }
195
196            let SignedBlock {
197                block,
198                justifications,
199            } = signed_block;
200            let (header, extrinsics) = block.deconstruct();
201            let hash = header.hash();
202
203            blocks_to_import.push(IncomingBlock {
204                hash,
205                header: Some(header),
206                body: Some(extrinsics),
207                indexed_body: None,
208                justifications,
209                origin: None,
210                allow_missing_state: false,
211                import_existing: false,
212                state: None,
213                skip_execution: false,
214            });
215
216            imported_blocks += 1;
217
218            if imported_blocks % 1000 == 0 {
219                debug!("Adding block {} from DSN to the import queue", block_number);
220            }
221        }
222
223        if !blocks_to_import.is_empty() {
224            // Import queue handles verification and importing it into the client
225            import_queue_service.import_blocks(BlockOrigin::NetworkInitialSync, blocks_to_import);
226        }
227
228        *last_processed_segment_index = segment_index;
229    }
230
231    Ok(imported_blocks)
232}
