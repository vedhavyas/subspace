1//! This crate provides a preprocessor for the domain block, which is used to construct
2//! domain extrinsics from the consensus block.
3//!
4//! The workflow is as follows:
5//! 1. Extract domain-specific bundles from the consensus block.
6//! 2. Compile the domain bundles into a list of extrinsics.
7//! 3. Shuffle the extrisnics using the seed from the consensus chain.
8//! 4. Filter out the invalid xdm extrinsics.
9//! 5. Push back the potential new domain runtime extrisnic.
10

42type DomainBlockElements<CBlock> = (Vec<<CBlock as BlockT>::Extrinsic>, Randomness);
43
44/// A wrapper indicating a valid bundle contents, or an invalid bundle reason.
45enum BundleValidity<Extrinsic> {
46    /// A valid bundle contents.
47    Valid(Vec<Extrinsic>),
48    /// An invalid bundle reason.
49    Invalid(InvalidBundleType),
50}

52/// Extracts the raw materials for building a new domain block from the primary block.
53fn prepare_domain_block_elements<Block, CBlock, CClient>(
54    consensus_client: &CClient,
55    block_hash: CBlock::Hash,
56) -> sp_blockchain::Result<DomainBlockElements<CBlock>>
57where
58    Block: BlockT,
59    CBlock: BlockT,
60    CClient: HeaderBackend<CBlock> + BlockBackend<CBlock> + ProvideRuntimeApi<CBlock> + Send + Sync,
61    CClient::Api: DomainsApi<CBlock, Block::Header>,
62{
63    let extrinsics = consensus_client.block_body(block_hash)?.ok_or_else(|| {
64        sp_blockchain::Error::Backend(format!("BlockBody of {block_hash:?} unavailable"))
65    })?;
66
67    let shuffling_seed = consensus_client
68        .runtime_api()
69        .extrinsics_shuffling_seed(block_hash)?;
70
71    Ok((extrinsics, shuffling_seed))
72}

154    pub fn preprocess_consensus_block(
155        &self,
156        consensus_block_hash: CBlock::Hash,
157        parent_domain_block: (Block::Hash, NumberFor<Block>),
158    ) -> sp_blockchain::Result<Option<PreprocessResult<Block>>> {
159        let (primary_extrinsics, shuffling_seed) = prepare_domain_block_elements::<Block, CBlock, _>(
160            &*self.consensus_client,
161            consensus_block_hash,
162        )?;
163
164        let bundles = self
165            .consensus_client
166            .runtime_api()
167            .extract_successful_bundles(consensus_block_hash, self.domain_id, primary_extrinsics)?;
168
169        if bundles.is_empty()
170            && !is_runtime_upgraded::<_, _, Block>(
171                &self.consensus_client,
172                consensus_block_hash,
173                self.domain_id,
174            )?
175        {
176            return Ok(None);
177        }
178
179        let tx_range = self
180            .consensus_client
181            .runtime_api()
182            .domain_tx_range(consensus_block_hash, self.domain_id)?;
183
184        let (inboxed_bundles, extrinsics) = self.compile_bundles_to_extrinsics(
185            bundles,
186            tx_range,
187            parent_domain_block,
188            consensus_block_hash,
189        )?;
190
191        let extrinsics =
192            deduplicate_and_shuffle_extrinsics::<Block::Extrinsic>(extrinsics, shuffling_seed);
193
194        Ok(Some(PreprocessResult {
195            extrinsics,
196            bundles: inboxed_bundles,
197        }))
198    }

200    /// Filter out the invalid bundles first and then convert the remaining valid ones to
201    /// a list of extrinsics.
202    #[allow(clippy::type_complexity)]
203    fn compile_bundles_to_extrinsics(
204        &self,
205        bundles: OpaqueBundles<CBlock, Block::Header, Balance>,
206        tx_range: U256,
207        (parent_domain_hash, parent_domain_number): (Block::Hash, NumberFor<Block>),
208        at_consensus_hash: CBlock::Hash,
209    ) -> sp_blockchain::Result<(
210        Vec<InboxedBundle<Block::Hash>>,
211        Vec<(Option<AccountId>, Block::Extrinsic)>,
212    )> {
213        let mut inboxed_bundles = Vec::with_capacity(bundles.len());
214        let mut valid_extrinsics = Vec::new();
215
216        let runtime_api = self.client.runtime_api();
217        for bundle in bundles {
218            // For the honest operator the validity of the extrinsic of the bundle is committed
219            // to (or say verified against) the receipt that is submitted with the bundle, the
220            // consensus runtime should only accept the bundle if the receipt is derived from
221            // the parent domain block. If it is not then either there is a bug in the consensus
222            // runtime (for validating the bundle) or in the domain client (for finding the parent
223            // domain block).
224            //
225            // NOTE: The receipt's `domain_block_number` is verified by the consensus runtime while
226            // the `domain_block_hash` is not (which is take care of by the fraud proof) so we can't
227            // check the parent domain block hash here.
228            if bundle.receipt().domain_block_number != parent_domain_number {
229                return Err(sp_blockchain::Error::RuntimeApiError(
230                    ApiError::Application(
231                        format!(
232                            "Unexpected bundle in consensus block: {:?}, something must be wrong",
233                            at_consensus_hash
234                        )
235                        .into(),
236                    ),
237                ));
238            }
239
240            let extrinsic_root = bundle.extrinsics_root();
241            match self.check_bundle_validity(
242                &bundle,
243                &tx_range,
244                (parent_domain_hash, parent_domain_number),
245                at_consensus_hash,
246            )? {
247                BundleValidity::Valid(extrinsics) => {
248                    let extrinsics: Vec<_> = match runtime_api
249                        .extract_signer(parent_domain_hash, extrinsics)
250                    {
251                        Ok(res) => res,
252                        Err(e) => {
253                            tracing::error!(error = ?e, "Error at calling runtime api: extract_signer");
254                            return Err(e.into());
255                        }
256                    };
257                    let bundle_digest: Vec<_> = extrinsics
258                        .iter()
259                        .map(|(signer, tx)| {
260                            (
261                                signer.clone(),
262                                ExtrinsicDigest::new::<LayoutV1<HeaderHashingFor<Block::Header>>>(
263                                    tx.encode(),
264                                ),
265                            )
266                        })
267                        .collect();
268                    inboxed_bundles.push(InboxedBundle::valid(
269                        HeaderHashingFor::<Block::Header>::hash_of(&bundle_digest),
270                        extrinsic_root,
271                    ));
272                    valid_extrinsics.extend(extrinsics);
273                }
274                BundleValidity::Invalid(invalid_bundle_type) => {
275                    inboxed_bundles
276                        .push(InboxedBundle::invalid(invalid_bundle_type, extrinsic_root));
277                }
278            }
279        }
280
281        Ok((inboxed_bundles, valid_extrinsics))
282    }

284    fn check_bundle_validity(
285        &self,
286        bundle: &OpaqueBundle<NumberFor<CBlock>, CBlock::Hash, Block::Header, Balance>,
287        tx_range: &U256,
288        (parent_domain_hash, parent_domain_number): (Block::Hash, NumberFor<Block>),
289        at_consensus_hash: CBlock::Hash,
290    ) -> sp_blockchain::Result<BundleValidity<Block::Extrinsic>> {
291        let bundle_vrf_hash =
292            U256::from_be_bytes(*bundle.sealed_header.header.proof_of_election.vrf_hash());
293
294        let mut extrinsics = Vec::with_capacity(bundle.extrinsics.len());
295        let mut estimated_bundle_weight = Weight::default();
296
297        let runtime_api = self.client.runtime_api();
298        let consensus_runtime_api = self.consensus_client.runtime_api();
299
300        // Check the validity of each extrinsic
301        //
302        // NOTE: for each extrinsic the checking order must follow `InvalidBundleType::checking_order`
303        for (index, opaque_extrinsic) in bundle.extrinsics.iter().enumerate() {
304            let decode_result =
305                runtime_api.decode_extrinsic(parent_domain_hash, opaque_extrinsic.clone())?;
306            let extrinsic = match decode_result {
307                Ok(extrinsic) => extrinsic,
308                Err(err) => {
309                    tracing::error!(
310                        ?opaque_extrinsic,
311                        ?err,
312                        "Undecodable extrinsic in bundle({})",
313                        bundle.hash()
314                    );
315                    return Ok(BundleValidity::Invalid(InvalidBundleType::UndecodableTx(
316                        index as u32,
317                    )));
318                }
319            };
320
321            let is_within_tx_range = runtime_api.is_within_tx_range(
322                parent_domain_hash,
323                &extrinsic,
324                &bundle_vrf_hash,
325                tx_range,
326            )?;
327
328            if !is_within_tx_range {
329                return Ok(BundleValidity::Invalid(InvalidBundleType::OutOfRangeTx(
330                    index as u32,
331                )));
332            }
333
334            // Check if this extrinsic is an inherent extrinsic.
335            // If so, this is an invalid bundle since these extrinsics should not be included in the
336            // bundle. Extrinsic is always decodable due to the check above.
337            if runtime_api.is_inherent_extrinsic(parent_domain_hash, &extrinsic)? {
338                return Ok(BundleValidity::Invalid(
339                    InvalidBundleType::InherentExtrinsic(index as u32),
340                ));
341            }
342
343            if let Some(xdm_mmr_proof) =
344                runtime_api.extract_xdm_mmr_proof(parent_domain_hash, &extrinsic)?
345            {
346                let ConsensusChainMmrLeafProof {
347                    opaque_mmr_leaf,
348                    proof,
349                    ..
350                } = xdm_mmr_proof;
351
352                if consensus_runtime_api
353                    .verify_proof(at_consensus_hash, vec![opaque_mmr_leaf], proof)?
354                    .is_err()
355                {
356                    return Ok(BundleValidity::Invalid(InvalidBundleType::InvalidXDM(
357                        index as u32,
358                    )));
359                }
360            }
361
362            // Using one instance of runtime_api throughout the loop in order to maintain context
363            // between them.
364            // Using `check_extrinsics_and_do_pre_dispatch` instead of `check_transaction_validity`
365            // to maintain side-effect in the storage buffer.
366            let is_legal_tx = runtime_api
367                .check_extrinsics_and_do_pre_dispatch(
368                    parent_domain_hash,
369                    vec![extrinsic.clone()],
370                    parent_domain_number,
371                    parent_domain_hash,
372                )?
373                .is_ok();
374
375            if !is_legal_tx {
376                return Ok(BundleValidity::Invalid(InvalidBundleType::IllegalTx(
377                    index as u32,
378                )));
379            }
380
381            let tx_weight = runtime_api.extrinsic_weight(parent_domain_hash, &extrinsic)?;
382            estimated_bundle_weight = estimated_bundle_weight.saturating_add(tx_weight);
383
384            extrinsics.push(extrinsic);
385        }
386
387        if estimated_bundle_weight != bundle.estimated_weight() {
388            return Ok(BundleValidity::Invalid(
389                InvalidBundleType::InvalidBundleWeight,
390            ));
391        }
392
393        Ok(BundleValidity::Valid(extrinsics))
394    }

397#[cfg(test)]
398mod tests {
399    use sp_domains::extrinsics::shuffle_extrinsics;
400    use sp_keyring::sr25519::Keyring;
401    use sp_runtime::traits::{BlakeTwo256, Hash as HashT};
402    use subspace_core_primitives::Randomness;
403
404    #[test]
405    fn shuffle_extrinsics_should_work() {
406        let alice = Keyring::Alice.to_account_id();
407        let bob = Keyring::Bob.to_account_id();
408        let charlie = Keyring::Charlie.to_account_id();
409
410        let extrinsics = vec![
411            (Some(alice.clone()), 10),
412            (None, 100),
413            (Some(bob.clone()), 1),
414            (Some(bob), 2),
415            (Some(charlie.clone()), 30),
416            (Some(alice.clone()), 11),
417            (Some(charlie), 31),
418            (None, 101),
419            (None, 102),
420            (Some(alice), 12),
421        ];
422
423        let dummy_seed = Randomness::from(BlakeTwo256::hash_of(&[1u8; 64]).to_fixed_bytes());
424        let shuffled_extrinsics = shuffle_extrinsics(extrinsics, dummy_seed);
425
426        assert_eq!(
427            shuffled_extrinsics,
428            vec![100, 30, 10, 1, 11, 101, 31, 12, 102, 2]
429        );
430    }
