288impl domain_pallet_executive::ExtrinsicStorageFees<Runtime> for ExtrinsicStorageFees {
289    fn extract_signer(xt: UncheckedExtrinsic) -> (Option<AccountId>, DispatchInfo) {
290        let dispatch_info = xt.get_dispatch_info();
291        let lookup = frame_system::ChainContext::<Runtime>::default();
292        let maybe_signer = extract_signer_inner(&xt, &lookup).and_then(|res| res.ok());
293        (maybe_signer, dispatch_info)
294    }
295
296    fn on_storage_fees_charged(
297        charged_fees: Balance,
298        tx_size: u32,
299    ) -> Result<(), TransactionValidityError> {
300        let consensus_storage_fee = BlockFees::consensus_chain_byte_fee()
301            .checked_mul(Balance::from(tx_size))
302            .ok_or(InvalidTransaction::Custom(ERR_BALANCE_OVERFLOW))?;
303
304        let (paid_consensus_storage_fee, paid_domain_fee) = if charged_fees <= consensus_storage_fee
305        {
306            (charged_fees, Zero::zero())
307        } else {
308            (consensus_storage_fee, charged_fees - consensus_storage_fee)
309        };
310
311        BlockFees::note_consensus_storage_fee(paid_consensus_storage_fee);
312        BlockFees::note_domain_execution_fee(paid_domain_fee);
313        Ok(())
314    }
315}

345impl sp_subspace_mmr::MmrProofVerifier<MmrHash, NumberFor<Block>, Hash> for MmrProofVerifier {
346    fn verify_proof_and_extract_leaf(
347        mmr_leaf_proof: ConsensusChainMmrLeafProof<NumberFor<Block>, Hash, MmrHash>,
348    ) -> Option<MmrLeaf<ConsensusBlockNumber, ConsensusBlockHash>> {
349        let ConsensusChainMmrLeafProof {
350            consensus_block_number,
351            opaque_mmr_leaf: opaque_leaf,
352            proof,
353            ..
354        } = mmr_leaf_proof;
355
356        if !is_consensus_block_finalized(consensus_block_number) {
357            return None;
358        }
359
360        let leaf: MmrLeaf<ConsensusBlockNumber, ConsensusBlockHash> =
361            opaque_leaf.into_opaque_leaf().try_decode()?;
362
363        verify_mmr_proof(vec![EncodableOpaqueLeaf::from_leaf(&leaf)], proof.encode())
364            .then_some(leaf)
365    }

529fn is_xdm_mmr_proof_valid(ext: &<Block as BlockT>::Extrinsic) -> Option<bool> {
530    match &ext.function {
531        RuntimeCall::Messenger(pallet_messenger::Call::relay_message { msg })
532        | RuntimeCall::Messenger(pallet_messenger::Call::relay_message_response { msg }) => {
533            let ConsensusChainMmrLeafProof {
534                consensus_block_number,
535                opaque_mmr_leaf,
536                proof,
537                ..
538            } = msg.proof.consensus_mmr_proof();
539
540            if !is_consensus_block_finalized(consensus_block_number) {
541                return Some(false);
542            }
543
544            Some(verify_mmr_proof(vec![opaque_mmr_leaf], proof.encode()))
545        }
546        _ => None,
547    }
548}

571fn extract_signer_inner<Lookup>(
572    ext: &UncheckedExtrinsic,
573    lookup: &Lookup,
574) -> Option<Result<AccountId, TransactionValidityError>>
575where
576    Lookup: sp_runtime::traits::Lookup<Source = Address, Target = AccountId>,
577{
578    match &ext.preamble {
579        Preamble::Bare(_) | Preamble::General(_, _) => None,
580        Preamble::Signed(address, _, _) => {
581            Some(lookup.lookup(address.clone()).map_err(|e| e.into()))
582        }
583    }
584}

586pub fn extract_signer(
587    extrinsics: Vec<UncheckedExtrinsic>,
588) -> Vec<(Option<opaque::AccountId>, UncheckedExtrinsic)> {
589    let lookup = frame_system::ChainContext::<Runtime>::default();
590
591    extrinsics
592        .into_iter()
593        .map(|extrinsic| {
594            let maybe_signer =
595                extract_signer_inner(&extrinsic, &lookup).and_then(|account_result| {
596                    account_result.ok().map(|account_id| account_id.encode())
597                });
598            (maybe_signer, extrinsic)
599        })
600        .collect()
601}

603fn extrinsic_era(extrinsic: &<Block as BlockT>::Extrinsic) -> Option<Era> {
604    match &extrinsic.preamble {
605        Preamble::Bare(_) | Preamble::General(_, _) => None,
606        Preamble::Signed(_, _, extra) => Some(extra.4 .0),
607    }

635        // for unsigned extrinsic, nonce check will be skipped
636        // so set a default value
637        frame_system::CheckNonce::<Runtime>::from(0u32.into()),
638        domain_check_weight::CheckWeight::<Runtime>::new(),
639        // for unsigned extrinsic, transaction fee check will be skipped
640        // so set a default value
641        pallet_transaction_payment::ChargeTransactionPayment::<Runtime>::from(0u128),
642        pallet_messenger::extensions::MessengerExtension::<Runtime>::new(),
643    );
644
645    UncheckedExtrinsic::new_transaction(call, extra)
646}

659fn check_transaction_and_do_pre_dispatch_inner(
660    uxt: &<Block as BlockT>::Extrinsic,
661) -> Result<(), TransactionValidityError> {
662    let lookup = frame_system::ChainContext::<Runtime>::default();
663
664    let xt = uxt.clone().check(&lookup)?;
665
666    let dispatch_info = xt.get_dispatch_info();
667
668    if dispatch_info.class == DispatchClass::Mandatory {
669        return Err(InvalidTransaction::MandatoryValidation.into());
670    }
671
672    let encoded_len = uxt.encoded_size();
673
674    // We invoke `pre_dispatch` in addition to `validate_transaction`(even though the validation is almost same)
675    // as that will add the side effect of SignedExtension in the storage buffer
676    // which would help to maintain context across multiple transaction validity check against same
677    // runtime instance.
678    match xt.format {
679        ExtrinsicFormat::General(extension_version, extra) => {
680            let origin = RuntimeOrigin::none();
681            <SignedExtra as DispatchTransaction<RuntimeCall>>::validate_and_prepare(
682                extra,
683                origin,
684                &xt.function,
685                &dispatch_info,
686                encoded_len,
687                extension_version,
688            )
689            .map(|_| ())
690        }
691        // signed transaction
692        ExtrinsicFormat::Signed(account_id, extra) => {
693            let custom_extra: CustomSignedExtra = (
694                extra.0,
695                extra.1,
696                extra.2,
697                extra.3,
698                extra.4,
699                extra.5,
700                extra.6.clone(),
701                extra.7,
702                pallet_messenger::extensions::MessengerTrustedMmrExtension::<Runtime>::new(),
703            );
704            let origin = RuntimeOrigin::signed(account_id);
705            <CustomSignedExtra as DispatchTransaction<RuntimeCall>>::validate_and_prepare(
706                custom_extra,
707                origin,
708                &xt.function,
709                &dispatch_info,
710                encoded_len,
711                // default extension version define here -
712                // https://github.com/paritytech/polkadot-sdk/blob/master/substrate/primitives/runtime/src/generic/checked_extrinsic.rs#L37
713                0,
714            )
715            .map(|_| ())
716        }
717        // unsigned transaction
718        ExtrinsicFormat::Bare => {
719            Runtime::pre_dispatch(&xt.function).map(|_| ())?;
720            <SignedExtra as TransactionExtension<RuntimeCall>>::bare_validate_and_prepare(
721                &xt.function,
722                &dispatch_info,
723                encoded_len,
724            )
725            .map(|_| ())
726        }
727    }
728}
