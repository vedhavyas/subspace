27/// Build the network service, the network status sinks and an RPC sender.
28#[allow(clippy::type_complexity)]
29#[expect(clippy::result_large_err, reason = "Comes from Substrate")]
30pub fn build_network<Block, Net, TxPool, IQ, Client, CBlock, CClient, DomainHeader>(
31    params: BuildNetworkParams<Block, Net, TxPool, IQ, Client>,
32    consensus_client: Arc<CClient>,
33) -> Result<
34    (
35        Arc<dyn sc_network::service::traits::NetworkService>,
36        TracingUnboundedSender<sc_rpc::system::Request<Block>>,
37        sc_network_transactions::TransactionsHandlerController<<Block as BlockT>::Hash>,
38        NetworkStarter,
39        Arc<SyncingService<Block>>,
40        NetworkServiceHandle,
41        Arc<dyn BlockDownloader<Block>>,
42    ),
43    Error,
44>
45where
46    Block: BlockT<Header = DomainHeader>,
47    CBlock: BlockT,
48    Client: ProvideRuntimeApi<Block>
49        + HeaderMetadata<Block, Error = sp_blockchain::Error>
50        + Chain<Block>
51        + BlockBackend<Block>
52        + BlockIdTo<Block, Error = sp_blockchain::Error>
53        + ProofProvider<Block>
54        + HeaderBackend<Block>
55        + BlockchainEvents<Block>
56        + AuxStore
57        + 'static,
58    TxPool: TransactionPool<Block = Block, Hash = <Block as BlockT>::Hash> + 'static,
59    IQ: ImportQueue<Block> + 'static,
60    Net: NetworkBackend<Block, <Block as BlockT>::Hash>,
61    CClient: ProvideRuntimeApi<CBlock>
62        + BlockBackend<CBlock>
63        + ProofProvider<CBlock>
64        + HeaderBackend<CBlock>
65        + Send
66        + Sync
67        + 'static,
68    CClient::Api: DomainsApi<CBlock, DomainHeader>,
69    DomainHeader: Header,
70{
71    let BuildNetworkParams {
72        config,
73        mut net_config,
74        client,
75        transaction_pool,
76        spawn_handle,
77        import_queue,
78        block_announce_validator_builder,
79        warp_sync_config,
80        block_relay,
81        metrics,
82    } = params;
83    let fork_id = config.chain_spec.fork_id();
84
85    let block_announce_validator = if let Some(f) = block_announce_validator_builder {
86        f(client.clone())
87    } else {
88        Box::new(DefaultBlockAnnounceValidator)
89    };
90
91    // "Last confirmed domain block execution receipt" request handler
92    {
93        let num_peer_hint = net_config.network_config.default_peers_set_num_full as usize
94            + net_config
95                .network_config
96                .default_peers_set
97                .reserved_nodes
98                .len();
99
100        let (handler, protocol_config) = LastDomainBlockERRequestHandler::new::<Net>(
101            fork_id,
102            consensus_client,
103            client.clone(),
104            num_peer_hint,
105        );
106        spawn_handle.spawn(
107            "last-domain-execution-receipt-request-handler",
108            Some("networking"),
109            handler.run(),
110        );
111
112        net_config.add_request_response_protocol(protocol_config);
113    }
114
115    let network_service_provider = NetworkServiceProvider::new();
116    let protocol_id = config.protocol_id();
117    let metrics_registry = config
118        .prometheus_config
119        .as_ref()
120        .map(|config| &config.registry);
121
122    let block_downloader = match block_relay {
123        Some(params) => {
124            let BlockRelayParams {
125                mut server,
126                downloader,
127                request_response_config,
128            } = params;
129
130            net_config.add_request_response_protocol(request_response_config);
131
132            spawn_handle.spawn("block-request-handler", Some("networking"), async move {
133                server.run().await;
134            });
135
136            downloader
137        }
138        None => build_default_block_downloader(
139            &protocol_id,
140            fork_id,
141            &mut net_config,
142            network_service_provider.handle(),
143            Arc::clone(&client),
144            config.network.default_peers_set.in_peers as usize
145                + config.network.default_peers_set.out_peers as usize,
146            &spawn_handle,
147        ),
148    };
149
150    let syncing_strategy = build_polkadot_syncing_strategy(
151        protocol_id.clone(),
152        fork_id,
153        &mut net_config,
154        warp_sync_config,
155        block_downloader.clone(),
156        client.clone(),
157        &spawn_handle,
158        metrics_registry,
159    )?;
160
161    let (syncing_engine, sync_service, block_announce_config) = SyncingEngine::new(
162        Roles::from(&config.role),
163        Arc::clone(&client),
164        metrics_registry,
165        metrics.clone(),
166        &net_config,
167        protocol_id.clone(),
168        fork_id,
169        block_announce_validator,
170        syncing_strategy,
171        network_service_provider.handle(),
172        import_queue.service(),
173        net_config.peer_store_handle(),
174        config.network.force_synced,
175    )?;
176
177    spawn_handle.spawn_blocking("syncing", None, syncing_engine.run());
178
179    let network_service_handle = network_service_provider.handle();
180    build_network_advanced(BuildNetworkAdvancedParams {
181        role: config.role,
182        protocol_id,
183        fork_id,
184        ipfs_server: config.network.ipfs_server,
185        announce_block: config.announce_block,
186        net_config,
187        client,
188        transaction_pool,
189        spawn_handle,
190        import_queue,
191        sync_service,
192        block_announce_config,
193        network_service_provider,
194        metrics_registry,
195        metrics,
196    })
197    .map(
198        |(network_service, system_rpc_tx, tx_handler_controller, network_starter, sync_service)| {
199            (
200                network_service,
201                system_rpc_tx,
202                tx_handler_controller,
203                network_starter,
204                sync_service,
205                network_service_handle,
206                block_downloader,
207            )
208        },
209    )
210}
