26// The slow log threshold for consensus block preprocessing
27const SLOW_PREPROCESS_MILLIS: u64 = 500;
28
29// The slow log threshold for domain block execution: `reference_duration_ms * 1.2 + 200ms`,
30// where `reference_duration_ms * 0.2` as buffer of the slow extrinsic execution (i.e. slower
31// machine than the reference machine) and 200ms as buffer of the rest of the processing.
32fn slow_domain_block_execution_threshold(reference_duration_ms: u64) -> u64 {
33    reference_duration_ms + (reference_duration_ms / 5) + 200
34}
35
36// The slow log threshold for post domain block execution
37const SLOW_POST_DOMAIN_BLOCK_EXECUTION_MILLIS: u64 = 250;

96impl<Block, CBlock, Client> domain_block_preprocessor::ValidateReceipt<Block, CBlock>
97    for ReceiptValidator<Client>
98where
99    Block: BlockT,
100    CBlock: BlockT,
101    Client: AuxStore,
102{
103    fn validate_receipt(
104        &self,
105        receipt: &ExecutionReceiptFor<Block, CBlock>,
106    ) -> sp_blockchain::Result<ReceiptValidity> {
107        // Skip genesis receipt as it has been already verified by the consensus chain.
108        if receipt.domain_block_number.is_zero() {
109            return Ok(ReceiptValidity::Valid);
110        }
111
112        let consensus_block_hash = receipt.consensus_block_hash;
113        let _local_receipt = crate::aux_schema::load_execution_receipt::<_, Block, CBlock>(
114            &*self.client,
115            consensus_block_hash,
116        )?
117        .ok_or_else(|| {
118            sp_blockchain::Error::Backend(format!(
119                "Receipt for consensus block {consensus_block_hash} not found"
120            ))
121        })?;
122
123        Ok(ReceiptValidity::Valid)
124    }
125}

188    // TODO: Handle the returned error properly, ref to https://github.com/autonomys/subspace/pull/695#discussion_r926721185
189    pub(crate) async fn process_bundles(
190        self,
191        consensus_block_info: (CBlock::Hash, NumberFor<CBlock>, bool),
192    ) -> sp_blockchain::Result<()> {
193        let (consensus_block_hash, consensus_block_number, is_new_best) = consensus_block_info;
194
195        // Skip processing the blocks of the non-canonical chain, these blocks will be processed if
196        // the chain becomes canonical later
197        if !is_new_best {
198            return Ok(());
199        }
200
201        tracing::debug!(
202            "Processing consensus block #{consensus_block_number},{consensus_block_hash}"
203        );
204
205        let maybe_pending_consensus_blocks = self
206            .domain_block_processor
207            .pending_imported_consensus_blocks(consensus_block_hash, consensus_block_number)?;
208
209        if let Some(PendingConsensusBlocks {
210            initial_parent,
211            consensus_imports,
212        }) = maybe_pending_consensus_blocks
213        {
214            tracing::trace!(
215                ?initial_parent,
216                ?consensus_imports,
217                "Pending consensus blocks to process"
218            );
219
220            let mut domain_parent = initial_parent;
221
222            for consensus_info in consensus_imports {
223                if let Some(next_domain_parent) = self
224                    .process_bundles_at((consensus_info.hash, consensus_info.number), domain_parent)
225                    .await?
226                {
227                    domain_parent = next_domain_parent;
228                }
229            }
230
231            // The domain branch driving from the best consensus branch should also be the best domain branch even
232            // if it is no the longest domain branch. Thus re-import the tip of the best domain branch to make it
233            // the new best block if it isn't.
234            //
235            // Note: this may cause the best domain fork switch to a shorter fork or in some case the best domain
236            // block become the ancestor block of the current best block.
237            let domain_tip = domain_parent.0;
238            if self.client.info().best_hash != domain_tip {
239                let header = self.client.header(domain_tip)?.ok_or_else(|| {
240                    sp_blockchain::Error::Backend(format!("Header for #{:?} not found", domain_tip))
241                })?;
242                let block_origin = if self
243                    .domain_block_processor
244                    .domain_sync_oracle
245                    .is_major_syncing()
246                {
247                    // The domain block is derived from the consensus block, if the consensus chain is
248                    // in major sync then we should also consider the domain block is `NetworkInitialSync`
249                    BlockOrigin::NetworkInitialSync
250                } else {
251                    BlockOrigin::Own
252                };
253                let block_import_params = {
254                    let mut import_block = BlockImportParams::new(block_origin, header);
255                    import_block.import_existing = true;
256                    import_block.fork_choice = Some(ForkChoiceStrategy::Custom(true));
257                    import_block.state_action = StateAction::Skip;
258                    import_block
259                };
260                self.domain_block_processor
261                    .import_domain_block(block_import_params)
262                    .await?;
263                assert_eq!(domain_tip, self.client.info().best_hash);
264
265                // finalize the latest confirmed domain block in the finalized Consensus block
266                self.finalize_domain_block((consensus_block_hash, consensus_block_number))?;
267            }
268
269            // Check the ER submitted to consensus chain and submit fraud proof if there is bad ER
270            // NOTE: this have to be done after the recorrect of the best domain fork happen above
271            self.domain_receipts_checker
272                .maybe_submit_fraud_proof(consensus_block_hash)?;
273        }
274
275        Ok(())
276    }

278    // finalize the domain block which is confirmed in the finalized consensus block.
279    fn finalize_domain_block(
280        &self,
281        consensus_block_info: (CBlock::Hash, NumberFor<CBlock>),
282    ) -> sp_blockchain::Result<()> {
283        if let Some(confirmed_consensus_block) = consensus_block_info
284            .1
285            .checked_sub(&self.confirmation_depth_k)
286        {
287            let runtime_api = self.consensus_client.runtime_api();
288            let confirmed_consensus_block_hash = self
289                .consensus_client
290                .hash(confirmed_consensus_block)?
291                .ok_or(sp_blockchain::Error::MissingHeader(format!(
292                    "Block Number: {}",
293                    confirmed_consensus_block
294                )))?;
295
296            let finalized_domain_block_number = self.client.info().finalized_number;
297            if let Some(confirmed_domain_block) = runtime_api
298                .latest_confirmed_domain_block(confirmed_consensus_block_hash, self.domain_id)?
299                && confirmed_domain_block.0 > finalized_domain_block_number
300            {
301                self.client
302                    .finalize_block(confirmed_domain_block.1, None, true)?;
303                info!("ðŸ”’ Finalized block: {:?}", confirmed_domain_block);
304            }
305        }
306        Ok(())
307    }

309    async fn process_bundles_at(
310        &self,
311        consensus_block_info: (CBlock::Hash, NumberFor<CBlock>),
312        parent_info: (Block::Hash, NumberFor<Block>),
313    ) -> sp_blockchain::Result<Option<(Block::Hash, NumberFor<Block>)>> {
314        let (consensus_block_hash, consensus_block_number) = consensus_block_info;
315        let (parent_hash, parent_number) = parent_info;
316        let start = Instant::now();
317
318        tracing::debug!(
319            "Building a new domain block from consensus block #{consensus_block_number},{consensus_block_hash} \
320            on top of parent block #{parent_number},{parent_hash}"
321        );
322
323        let maybe_preprocess_result = self
324            .domain_block_preprocessor
325            .preprocess_consensus_block(consensus_block_hash, (parent_hash, parent_number))?;
326
327        let preprocess_took = start.elapsed().as_millis();
328        if preprocess_took >= SLOW_PREPROCESS_MILLIS.into() {
329            tracing::warn!(
330                ?consensus_block_info,
331                "Slow consensus block preprocessing, took {preprocess_took}ms"
332            );
333        }
334
335        let Some(preprocess_result) = maybe_preprocess_result else {
336            tracing::debug!(
337                "Skip building new domain block, no bundles and runtime upgrade for this domain \
338                    in consensus block #{consensus_block_number:?},{consensus_block_hash}"
339            );
340            self.domain_block_processor
341                .on_consensus_block_processed(consensus_block_hash, None)?;
342            return Ok(None);
343        };
344
345        let inherent_digests = Digest {
346            logs: vec![DigestItem::consensus_block_info(consensus_block_hash)],
347        };
348
349        let domain_block_result = self
350            .domain_block_processor
351            .process_domain_block(
352                (consensus_block_hash, consensus_block_number),
353                (parent_hash, parent_number),
354                preprocess_result,
355                inherent_digests,
356            )
357            .await?;
358
359        let head_receipt_number = self
360            .consensus_client
361            .runtime_api()
362            .head_receipt_number(consensus_block_hash, self.domain_id)?;
363        assert!(
364            domain_block_result.header_number > head_receipt_number,
365            "Domain chain number must larger than the head number of the receipt chain \
366            (which is maintained on the consensus chain) by at least 1"
367        );
368
369        let built_block_info = (
370            domain_block_result.header_hash,
371            domain_block_result.header_number,
372        );
373
374        let block_execution_took = start.elapsed().as_millis().saturating_sub(preprocess_took);
375
376        let reference_block_execution_duration_ms = self
377            .client
378            .runtime_api()
379            .block_weight(domain_block_result.header_hash)?
380            .ref_time()
381            / WEIGHT_REF_TIME_PER_MILLIS;
382        if block_execution_took
383            >= slow_domain_block_execution_threshold(reference_block_execution_duration_ms).into()
384        {
385            tracing::warn!(
386                ?consensus_block_info,
387                ?built_block_info,
388                ?reference_block_execution_duration_ms,
389                "Slow domain block execution, took {block_execution_took}ms"
390            );
391        }
392
393        self.domain_block_processor
394            .on_consensus_block_processed(consensus_block_hash, Some(domain_block_result))?;
395
396        let post_block_execution_took = start
397            .elapsed()
398            .as_millis()
399            .saturating_sub(preprocess_took + block_execution_took);
400        if post_block_execution_took >= SLOW_POST_DOMAIN_BLOCK_EXECUTION_MILLIS.into() {
401            tracing::warn!(
402                ?consensus_block_info,
403                ?built_block_info,
404                "Slow post domain block execution, took {post_block_execution_took}ms"
405            );
406        }
407
408        Ok(Some(built_block_info))
409    }
